```rust
/// Salt prefixes — matched longest-first. Used to strip chemical form
/// from active ingredient names (e.g., "chlorhydrate de paracétamol" → "paracétamol").
pub static SALT_PREFIXES: &[&str] = &[
    // Match longest-first: "chlorhydrate de" before "chlorhy"
    "chlorhydrate de", "chlorhydrate d'", "chlorhy de", "chlorhy d'",
    "sulfate de", "sulfate d'",
    "malate de", "malate d'",
    "bromhydrate de", "bromhydrate d'",
    "tartrate de", "tartrate d'",
    "glycolate de",
    "nicotinate de",
    "phosphate de", "phosphate d'",
    "acétate de", "acétate d'",
    "borate de",
    "citrate de", "citrate d'",
    "glucoheptonate de",
    " gluconate de", "gluconate d'",
    "lactate de", "lactate d'",
    "maleate de", "maleate d'",
    "camphosulfonate de",
    "bésylate de", "bésylate d'",
    "sel de", "sel d'",
    "iodure de",
    "bromure de",
    "chlorure de", "chlorure d'",
    "nitrate de",
    "ranélate de",
];

/// Salt suffixes — matched iteratively until no change (multi-pass).
/// E.g., "paracetamol chlorhydrate monohydrate" → "paracetamol chlorhydrate" → "paracetamol"
pub static SALT_SUFFIXES: &[&str] = &[
    "chlorhydrate monohydrate", "chlorhydrate anhydre",
    "chlorhydrate",
    "sulfate anhydre", "sulfate",
    "malate", "bromhydrate",
    "tartrate", "glycolate",
    "base anhydre", "base",
    "sel sodique", "sel",
    "dihydrate", "trihydrate", "monohydrate", "anhydre",
    "sel de sodium",
    "chlorhydrate de sodium",
];

/// Strip salt prefixes and suffixes from active ingredient names.
/// Multi-pass for suffixes: strips one suffix, then re-checks until no change.
pub fn strip_salt(s: &str) -> String {
    let s = s.trim();
    // Strip prefix (check start of string, longest-first)
    let mut result = s.to_string();
    for prefix in SALT_PREFIXES.iter() {
        let prefix_upper = prefix.to_uppercase();
        let result_upper = result.to_uppercase();
        if result_upper.starts_with(&prefix_upper) {
            result = result[prefix.len()..].trim().to_string();
            break;
        }
    }
    // Multi-pass suffix stripping
    loop {
        let before = result.clone();
        for suffix in SALT_SUFFIXES.iter() {
            let suffix_upper = suffix.to_uppercase();
            let result_upper = result.to_uppercase();
            if result_upper.ends_with(&suffix_upper) {
                let end_pos = result_upper.len() - suffix_upper.len();
                result = result[..end_pos].trim().to_string();
            }
        }
        if result == before { break; }
    }
    result
}
```

```rust
/// Strip parenthetical salt annotations mid-string.
/// E.g., "Paracetamol (chlorhydrate)" → "Paracetamol"
/// Note: uses greedy `\([^)]*\)` — will not handle nested parens, which
/// do not appear in BDPM substance data. Keep it simple.
pub fn strip_parens(s: &str) -> String {
    static RE: LazyLock<Regex> = LazyLock::new(|| {
        Regex::new(r"\s*\([^)]*\)").unwrap()
    });
    RE.replace_all(s, "").trim().to_string()
}
```

### Task 2: Add post-import validation threshold checks

```rust
fn validate_thresholds(conn: &Connection) {
    // Ghost CIS threshold
    let ghost_cis: i64 = conn.query_row(
        "SELECT COUNT(*) FROM generic_groups WHERE is_orphan = 1", [], |r| r.get(0)
    ).unwrap_or(0);
    if ghost_cis > 3000 {
        tracing::warn!("Ghost CIS count {} exceeds threshold 3000 — possible format change", ghost_cis);
    }

    // Chemical ID cardinality
    let substance_count: i64 = conn.query_row(
        "SELECT COUNT(DISTINCT substance_code) FROM compositions", [], |r| r.get(0)
    ).unwrap_or(0);
    if !(2000..=3000).contains(&substance_count) {
        tracing::warn!("Substance code cardinality {} outside expected range 2000-3000", substance_count);
    }

    // Princeps coverage
    let princeps_groups: i64 = conn.query_row(
        "SELECT COUNT(DISTINCT group_id) FROM generic_groups WHERE type = 'reference'", [], |r| r.get(0)
    ).unwrap_or(0);
    if princeps_groups < 500 {
        tracing::warn!("Princeps groups {} below expected minimum 500", princeps_groups);
    }

    // Generic name coverage
    // Note: `substance_name_clean` column exists in schema.sql (line ~74).
    // It is populated via `normalize_compo()` which stores values[8]. First run
    // after Phase 05 will have low coverage — this is expected. Backfill via
    // UPDATE compositions SET substance_name_clean = strip_salt(substance_name_clean)
    // if needed, or wait for next ingest cycle.
    let coverage: f64 = conn.query_row(
        "SELECT COUNT(DISTINCT substance_name_clean) * 1.0 /
         NULLIF(COUNT(DISTINCT substance_code), 0) FROM compositions
         WHERE substance_name_clean IS NOT NULL",
        [], |r| r.get(0)
    ).unwrap_or(0.0);
    if coverage < 0.5 {
        tracing::warn!("Generic name coverage {:.1}% below 50% threshold (expected on first run)", coverage * 100.0);
    }

    // Temporal coherence: auth_date <= comm_date
    let date_coherence: i64 = conn.query_row(
        "SELECT COUNT(*) FROM presentations p
         JOIN drugs d ON p.cis = d.cis
         WHERE p.comm_date IS NOT NULL AND d.auth_date IS NOT NULL
         AND p.comm_date < d.auth_date", [], |r| r.get(0)
    ).unwrap_or(0);
    if date_coherence > 0 {
        tracing::warn!("Date coherence issues: {} presentations have comm_date before auth_date", date_coherence);
    }
}
```

```rust
/// Generate an FTS-searchable version of a drug name.
/// Strips diacritics, normalizes spaces, removes noise words.
pub static NOISE_WORDS: LazyLock<phf::Set<&'static str>> = LazyLock::new(|| {
    phf_set! {
        "de", "du", "la", "le", "les", "et", "ou", "en", "un", "une",
        "des", "aux", "au", "a", "l", "d",
    }
});

pub fn fts_normalize(s: &str) -> String {
    strip_diacritics(&s.to_lowercase())
        .split_whitespace()
        .filter(|w| !NOISE_WORDS.contains(w))
        .collect::<Vec<_>>()
        .join(" ")
}
```

### Task 4: Add rayon parallelization for bulk normalization
Files: Cargo.toml, src/normalize/mod.rs, src/import/mod.rs
Action: Add `rayon = "1.10"` to Cargo.toml. In `src/import/mod.rs`, replace the CIS_COMPO normalization block (the `filter_map(|line| normalize_row(...))` call for `CIS_COMPO_bdpm`) with a parallel version.

**Current pipeline order for CIS_COMPO_bdpm:**
1. Read raw rows from file
2. `filter_map(normalize_compo)` — produces `Vec<NormalizedRow>`
3. `dedup_compo()` — sequential HashSet dedup
4. `import_file()` — insert into DB

**New parallel order:**
1. Read raw rows from file (sequential)
2. **NEW: `into_par_iter().map(normalize_compo).collect()`** — parallel normalization
3. **Sort by index before dedup** — parallel output is unordered, dedup needs deterministic order
4. `dedup_compo()` — unchanged
5. `import_file()` — unchanged

```rust
use rayon::prelude::*;

// In the CIS_COMPO_bdpm block of run_ingest(), replace the sequential:
// let mut rows: Vec<_> = raw_lines.filter_map(|line| normalize_row(...)).collect();
// With:
let mut rows: Vec<_> = raw_lines
    .filter_map(|line| normalize_row(...))
    .collect();

// Parallel normalize and re-index by original line number
let (tx, rx) = std::sync::mpsc::channel();
raw_lines.into_iter().enumerate().collect::<Vec<_>>().into_par_iter().for_each(|(idx, raw)| {
    if let Some(normalized) = normalize_row(raw) {
        let _ = tx.send((idx, normalized));
    }
});
let mut parallel_rows: Vec<(usize, NormalizedRow)> = rx.into_iter().collect();
// Sort by original index to ensure deterministic dedup input order
parallel_rows.sort_by_key(|(idx, _)| *idx);
let rows: Vec<NormalizedRow> = parallel_rows.into_iter().map(|(_, r)| r).collect();
let rows = dedup_compo(rows); // dedup unchanged
```

**Important:** The parallel normalization requires collecting all raw rows into memory first (Vec), then distributing to parallel workers. For 32K rows this is acceptable. The `rx.recv()` loop collects all results, sorts by index, then passes to `dedup_compo()` as before. The dedup ordering is preserved.

Add a `#[cfg(test)]` test that verifies parallel output matches sequential output for a sample of 100 rows (determinism check).
