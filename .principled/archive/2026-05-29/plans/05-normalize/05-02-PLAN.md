## Decision: phf for Static Maps

Use `phf = "0.11"` for compile-time perfect-hash lookup tables. These maps are static (don't change between builds), small enough for compile-time generation, and `phf::phf_map!` produces zero-heap-allocation O(1) lookups. This is better than `Lazy<HashMap>` because the data is constant and known at compile time.

   pub fn canonicalize_lab(lab: &str) -> String {
       let normalized = lab.trim().to_uppercase();
       LAB_FAMILY_MAP.get(&normalized)
           .map(|s| s.to_string())
           .unwrap_or_else(|| {
               // Fallback: strip common suffixes and retry
               strip_lab_suffix(&normalized)
           })
   }

   fn strip_lab_suffix(lab: &str) -> String {
       // Strip corporate suffixes: LABORATOIRES, PHARMA, SAS, SARL, GMBH, LTD, France, etc.
       let suffixes = ["LABORATOIRES", "PHARMA", "SAS", "SARL", "GMBH", "LTD", "SA", "FRANCE", "HEALTH", "SANTE"];
       let mut result = lab.to_string();
       for suffix in suffixes {
           if let Some(pos) = result.strip_prefix(suffix) {
               result = pos.trim().to_string();
           }
           if let Some(pos) = result.strip_suffix(suffix) {
               result = pos.trim().to_string();
           }
       }
       if result.is_empty() { lab.to_string() } else { result }
   }

### Task 4: Add CIS_CPD prescription condition flags

```rust
use regex_lite::Regex;
use once_cell::sync::Lazy;
use std::collections::HashMap;

static CPD_PATTERNS: Lazy<HashMap<&'static str, Regex>> = Lazy::new(|| {
    HashMap::from([
        ("liste_i", Regex::new(r"(?i)\bliste\s*i\b").unwrap()),
        ("liste_ii", Regex::new(r"(?i)\bliste\s*ii\b").unwrap()),
        ("stupefiant", Regex::new(r"(?i)stupéfiant|stupefiant").unwrap()),
        ("hospitalier", Regex::new(r"(?i)hospitalier|hospitalière|rérve|hôpital").unwrap()),
        ("dentaire", Regex::new(r"(?i)dentaire").unwrap()),
        ("reserve_hopital", Regex::new(r"(?i)réserve\s*d'?hôpital|reste?\s*à?\s*l'?hopital").unwrap()),
    ])
});

#[derive(Default, Debug, Clone)]
pub struct CpdFlags {
    pub liste_i: bool,
    pub liste_ii: bool,
    pub stupefiant: bool,
    pub hospitalier: bool,
    pub dentaire: bool,
    pub reserve_hopital: bool,
}

impl CpdFlags {
    pub fn from_rule(rule: &str) -> Self {
        let upper = rule.to_uppercase();
        let mut flags = CpdFlags::default();
        for (flag_name, re) in CPD_PATTERNS.iter() {
            match *flag_name {
                "liste_i" => flags.liste_i = re.is_match(&upper),
                "liste_ii" => flags.liste_ii = re.is_match(&upper),
                "stupefiant" => flags.stupefiant = re.is_match(&upper),
                "hospitalier" => flags.hospitalier = re.is_match(&upper),
                "dentaire" => flags.dentaire = re.is_match(&upper),
                "reserve_hopital" => flags.reserve_hopital = re.is_match(&upper),
                _ => {}
            }
        }
        flags
    }
}
```
