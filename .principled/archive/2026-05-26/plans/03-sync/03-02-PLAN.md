## Tasks

### Task 1: Weekly dispo-only sync
Files: `src/sync/dispo.rs`
Action:
```rust
pub fn sync_dispo_only(conn: &Connection, raw_dir: &Path, state: &mut StateStore) -> Result<ImportStats> {
    let file = BDPMFile::CIS_CIP_Dispo_Spec;
    let url = format!("{}{}", BDPM_URL, file.download_path());

    let bytes = fetcher.fetch(&url, raw_dir)?;
    let hash = blake3::hash(&bytes).to_hex().to_string();

    if !state.needs_update(file, &hash, bytes.len() as u64) {
        println!("Dispo: unchanged");
        return Ok(ImportStats::zero());
    }

    // Parse and import
    let rows = parse_file(&raw_dir.join(file.filename()), file)?;
    let normalized = normalize_rows(file, &rows)?;
    let stats = import_file(conn, file, &normalized)?;

    state.mark_updated(file, &hash, bytes.len() as u64);
    Ok(stats)
}
```

CLI command:
```rust

} => {
    let mut state = StateStore::load_or_create(&data_dir)?;
    let conn = init_db(&data_dir.join("bdpm.db"))?;

    println!("Dispo: {} rows, {} bad, {} skipped ({}ms)",
        stats.rows_imported, stats.bad_rows, stats.skipped_rows, stats.duration_ms);

}
```

### Task 2: GitHub Actions workflows
Files: `.github/workflows/monthly-sync.yml`, `.github/workflows/weekly-dispo.yml`
Action:

```yaml
# .github/workflows/monthly-sync.yml
name: Monthly BDPM Sync
on:
  schedule:
    - cron: '0 2 1 * *'  # 1st of month, 02:00 UTC
  workflow_dispatch:
    inputs:
      full:
        description: 'Force full reimport'
        required: false
        default: 'false'

jobs:
  sync:
    runs-on: ubuntu-latest
    permissions:
      contents: write
    steps:
      - uses: actions/checkout@v4
        with:
          # Fetch previous state from release
          ref: refs/heads/main

      - uses: dtolnay/rust-toolchain@stable

      - uses: Swatinem/rust-cache@v2

      - name: Download previous DB artifact
        if: github.event.inputs.full != 'true'
        run: gh release download latest --dir data --pattern "bdpm.db" || echo "No previous release"

      - name: Build and sync
        run: |
          cargo build --release
          ./target/release/bdpm-ingest import \
            --data-dir data \
            ${{ github.event.inputs.full == 'true' && '--full' || '' }}

      - name: WAL checkpoint
        run: sqlite3 data/bdpm.db "PRAGMA wal_checkpoint(TRUNCATE);"

      - name: Publish release
        uses: softprops/action-gh-release@v2
        with:
          tag_name: bdpm-${{ env.GITHUB_REF_NAME }}
          files: data/bdpm.db
```

### Task 3: Health check endpoint
Files: `src/api/health.rs`

```rust
pub async fn health(State(state): State<AppState>) -> Json<HealthResponse> {
    let db_fresh = tokio::task::spawn_blocking(move || {
        let conn = Connection::open(&state.db_path)?;
        let last_import: Option<String> = conn.prepare(
            "SELECT MAX(imported_at) FROM import_log WHERE status = 'success'"
        )?.query_row([], |row| row.get(0)).ok();

        let drug_count: i64 = conn.prepare(
            "SELECT COUNT(*) FROM drugs"
        )?.query_row([], |row| row.get(0)).unwrap_or(0);

        HealthResponse { last_import, drug_count, status: "ok" }
    }).await.unwrap();

    Json(db_fresh)
}
```
