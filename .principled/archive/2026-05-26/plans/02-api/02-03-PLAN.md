// GET /generic-groups — list all groups with CIS count
pub async fn list_generic_groups(
    State(state): State<AppState>,
) -> Json<Vec<GenericGroupList>> {
    tokio::task::spawn_blocking(move || {
        let conn = Connection::open(&state.db_path)?;
        let mut stmt = conn.prepare(
            "SELECT group_id, group_name, COUNT(cis) as cis_count
             FROM generic_groups
             GROUP BY group_id, group_name
             ORDER BY group_id"
        )?;
        let rows = stmt.query_map([], |row| Ok(GenericGroupList {
            group_id: row.get(0)?,
            group_name: row.get(1)?,
            cis_count: row.get(2)?,
        }))?.filter_map(|r| r.ok()).collect::<Vec<_>>();
        Ok(rows)
    }).await.unwrap()
}

// GET /generic-groups/:id — drugs in a group
pub async fn generic_group_detail(
    Path(group_id): Path<String>,
    State(state): State<AppState>,
) -> Json<Vec<GenericGroupMember>> {
    tokio::task::spawn_blocking(move || {
        let conn = Connection::open(&state.db_path)?;
        let mut stmt = conn.prepare(
            "SELECT g.cis, d.name, g.type, g.sort_order, g.is_orphan
             FROM generic_groups g
             LEFT JOIN drugs d ON g.cis = d.cis
             WHERE g.group_id = ?1
             ORDER BY g.sort_order, d.name"
        )?;
        let rows = stmt.query_map([&group_id], |row| Ok(GenericGroupMember {
            cis: row.get(0)?, name: row.get(1)?, type_: row.get(2)?,
            sort_order: row.get(3)?, is_orphan: row.get(4)?,
        }))?.filter_map(|r| r.ok()).collect::<Vec<_>>();
        Ok(rows)
    }).await.unwrap()
}

// GET /atc/:code — detail + child codes + drugs
pub async fn atc_detail(
    Path(code): Path<String>,
    State(state): State<AppState>,
) -> Json<AtcDetail> {
    tokio::task::spawn_blocking(move || {
        let conn = Connection::open(&state.db_path)?;

        // Get ATC info
        let atc: AtcCode = conn.prepare_row(
            &conn.prepare("SELECT atc_code, parent_1_char FROM atc_codes WHERE atc_code = ?1")?
        )?.query_row([&code], |row| Ok(AtcCode { atc_code: row.get(0)?, parent_char: row.get(1)? }))?;

        // Child codes (ATC codes starting with this prefix, longer than current)
        let child_codes = conn.prepare(
            "SELECT atc_code FROM atc_codes
             WHERE atc_code LIKE ?1 AND LENGTH(atc_code) = ?2"
        )?.query_map(params![format!("{}%", code), code.len() + 3], |row| {
            row.get::<_, String>(0)
        })?.filter_map(|r| r.ok()).collect::<Vec<_>>();

        // Drugs under this ATC (via mitm join)
        let drugs = conn.prepare(
            "SELECT d.cis, d.name FROM drugs d
             JOIN mitm m ON m.cis = d.cis
             WHERE m.atc_code = ?1
             LIMIT 50"
        )?.query_map([&code], |row| Ok(AtcDrug { cis: row.get(0)?, name: row.get(1)? }))?
         .filter_map(|r| r.ok()).collect::<Vec<_>>();

        Ok(AtcDetail { atc, children: child_codes, drugs })
    }).await.unwrap()
}
