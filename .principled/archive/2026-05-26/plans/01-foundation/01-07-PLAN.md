### Task 3: Write integration tests
Files: `tests/full_import.rs`
Action:
```rust
/// End-to-end test: full import into in-memory DB
#[test]
fn full_import_row_counts() {
    let conn = Connection::open_in_memory();
    init_db(&conn).unwrap();

    // Import all files
    let report = import_all(&conn, Path::new("raw"), &StateStore::default()).unwrap();

    assert_eq!(count(&conn, "drugs"), 15_848);
    assert_eq!(count(&conn, "presentations"), 20_903);
    assert_eq!(count(&conn, "compositions"), 27_609);
}

#[test]
fn orphan_counts_match_baseline() {
    // From BRIEF.md confirmed counts
    assert_eq!(smr_orphan_count(), 2_806);
    assert_eq!(asmr_orphan_count(), 1_567);
    assert_eq!(gen_orphan_count(), 2_503);
}

#[test]
fn dispo_check_constraint() {
    // No status_type outside 1-4
    let bad = query_single_int("SELECT COUNT(*) FROM availability WHERE status_type NOT IN (1,2,3,4)");
    assert_eq!(bad, 0);
}
```
Verify: `cargo test --test full_import` passes
Done: Integration tests written and passing

### Task 4: GitHub Actions CI scaffold
Files: `.github/workflows/ci.yml`
Action:
```yaml
name: CI
on: [push, pull_request]
jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-toolchain@stable
      - uses: Swatinem/rust-cache@v2
      - run: cargo test --all
      - run: cargo clippy -- -D warnings
      - name: Integration tests
        run: |
          cargo build --release
          ./target/release/bdpm-ingest import --full --data-dir data
          sqlite3 data/bdpm.db "SELECT COUNT(*) FROM drugs"  # must be 15848
```
Also create `.github/workflows/monthly-sync.yml` (skeleton):
```yaml
name: Monthly BDPM Sync
on:
  schedule:
    - cron: '0 2 1 * *'  # 1st of month, 02:00 UTC
  workflow_dispatch:
jobs:
  sync:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - run: cargo build --release
      - run: ./target/release/bdpm-ingest import --full --data-dir data
      - run: sqlite3 data/bdpm.db "PRAGMA wal_checkpoint(TRUNCATE);"
      - uses: softprops/action-gh-release@v2
        with:
          files: data/bdpm.db
```
Verify: `cargo test` passes locally
Done: CI workflow scaffolded
