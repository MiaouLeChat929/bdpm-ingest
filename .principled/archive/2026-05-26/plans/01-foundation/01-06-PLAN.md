### Task 3: Orchestrate the full pipeline
Files: `src/import/mod.rs`
Action:
```rust
pub fn run_import(data_dir: &Path, full: bool, file_filter: Option<&str>) -> Result<ImportReport> {
    // 1. Ensure data directory
    std::fs::create_dir_all(data_dir)?;

    // 2. Load or create state
    let mut state = StateStore::load_or_create(data_dir)?;

    // 3. Fetcher
    let fetcher = Fetcher::new();
    let raw_dir = data_dir.join("raw");
    std::fs::create_dir_all(&raw_dir)?;

    // 4. DB path
    let db_path = data_dir.join("bdpm.db");
    let conn = init_db(&db_path)?;

    // 5. Iterate BDPMFile, fetch + parse + import
    let order = [...]; // from 01-05
    let mut report = ImportReport::default();

    for file in &order {
        // Filter if --file flag set
        if let Some(f) = file_filter {
            if file.filename() != f { continue; }
        }

        // Check if changed
        let url = format!("{}{}", BASE_URL, file.download_path());
        let bytes = fetcher.fetch(&url, &raw_dir)?;
        let hash = blake3::hash(&bytes).to_hex().to_string();

        if !full && !state.needs_update(file, &hash, bytes.len() as u64) {
            println!("{}: unchanged ({})", file.filename(), &hash[..8]);
            continue;
        }

        // Parse
        let rows = parse_file(&raw_dir.join(file.filename()), file)?;
        let normalized = normalize_rows(file, &rows)?;

        // Import
        match import_file(&conn, file, &normalized) {
            Ok(stats) => {
                state.mark_updated(file, &hash, bytes.len() as u64);
                report.success(file, stats);
            }
            Err(e) => {
                report.failure(file, e);
            }
        }
    }

    state.save(&state_path(data_dir))?;
    Ok(report)
}
```
Verify: `bdpm-ingest import --full` completes without panic
Done: Full pipeline runs end-to-end
